  enabled: boolean;
  lastScanAt: string | null;
  // ...other fields
}

export class MoveTracker {
  private scanTimer: ReturnType<typeof setInterval> | null = null;
  private running = false;
  
  start(): void {
    if (!this.deps.enabled) return;
    this.stop();
    // Fire-and-forget initial scan
    this.runScan().catch(err => { ... });
    this.scanTimer = setInterval(() => { ... }, this.deps.scanIntervalMs);
  }
  
  stop(): void {
    if (this.scanTimer !== null) {
      clearInterval(this.scanTimer);
      this.scanTimer = null;
    }
  }
  
  getState(): MoveTrackerState { ... }
  async runScanForTest(): Promise<void> { return this.runScan(); }
  
  private async runScan(): Promise<void> {
    if (this.running || this.deps.client.state !== 'connected') return;
    this.running = true;
    try { ... }
    catch (err) {
      const code = (err as { code?: string })?.code;
      if (code === 'NoConnection' || code === 'ETIMEOUT') {
        this.deps.logger?.debug({ err }, '...');
      } else { throw err; }
    } finally { this.running = false; }
  }
}
