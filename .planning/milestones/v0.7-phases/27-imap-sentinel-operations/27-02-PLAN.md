```typescript
import type { ImapClient, AppendResponse } from '../imap/index.js';
import type { SentinelStore } from './store.js';
import type { FolderPurpose } from './format.js';
import { buildSentinelMessage } from './format.js';

export interface AppendSentinelResult {
  messageId: string;
  uid: number | undefined;
}

export async function appendSentinel(
  client: ImapClient,
  folder: string,
  purpose: FolderPurpose,
  store?: SentinelStore,
): Promise<AppendSentinelResult> {
  const msg = buildSentinelMessage({ folderPath: folder, folderPurpose: purpose });
  const result = await client.appendMessage(folder, msg.raw, msg.flags);
  if (store) {
    store.upsert({ messageId: msg.messageId, folderPath: folder, folderPurpose: purpose });
  }
  return { messageId: msg.messageId, uid: result.uid };
}
```

`findSentinel(client, folder, messageId)`: calls `client.searchByHeader(folder, 'X-Mail-Mgr-Sentinel', messageId)`, returns `uids.length > 0 ? uids[0] : undefined`.

`deleteSentinel(client, folder, uid, store?, messageId?)`: calls `client.deleteMessage(folder, uid)`. If store and messageId provided, calls `store.deleteByMessageId(messageId)`.

`runSentinelSelfTest(client, testFolder, logger)`: Full round-trip per D-04. Uses try/catch/finally per D-05. Self-test body text: 'Self-test sentinel — safe to delete'. On success: `logger.info('Sentinel self-test passed: SEARCH HEADER supported')`. On failure: `logger.warn(...)`. Returns boolean. Uses `appendSentinel` (without store) internally, then `findSentinel`, then cleanup with `deleteSentinel` in finally block. The `findSentinel` call uses the messageId from appendSentinel result. For cleanup, uses the UID from appendMessage result. If APPEND didn't return a UID (no UIDPLUS), falls back to `findSentinel` to get UID for cleanup.
