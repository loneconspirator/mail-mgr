   export function generateBehaviorDescription(rule: Rule): string {
     const parts: string[] = [];
     if (rule.match.sender) parts.push(`sender:${rule.match.sender}`);
     if (rule.match.recipient) parts.push(`recipient:${rule.match.recipient}`);
