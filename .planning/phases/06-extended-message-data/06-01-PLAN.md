export class ActivityLog {
  private db: Database.Database;
  constructor(dbPath: string) {
    this.db = new Database(dbPath);
    this.db.pragma('journal_mode = WAL');
    this.db.exec(SCHEMA);

  }

      this.db.exec(`ALTER TABLE activity ADD COLUMN source TEXT NOT NULL DEFAULT 'arrival'`);
    } catch { /* Column already exists */ }
    try {
      this.db.exec(`CREATE INDEX IF NOT EXISTS idx_activity_source ON activity(source)`);
    } catch { /* Index already exists */ }
    try {
      this.db.exec(`CREATE INDEX IF NOT EXISTS idx_activity_folder_success ON activity(folder, success)`);
    } catch { /* Index already exists */ }
  }
}

    **Create `src/log/migrations.ts`:**

    Define the Migration interface and migrations array per D-09:
    ```typescript
    import type Database from 'better-sqlite3';

    export interface Migration {
      version: string;       // Timestamp format: YYYYMMDD_NNN
      description: string;
      up: (db: Database.Database) => void;
    }
    ```

    Define the migrations array with one bootstrap migration (per D-10):
    - version: `'20260411_001'`
    - description: `'Bootstrap: source column and indexes'`
    - up function: Uses `db.pragma('table_info(activity)')` to check if 'source' column exists. If not, runs `ALTER TABLE activity ADD COLUMN source TEXT NOT NULL DEFAULT 'arrival'`. Then runs `CREATE INDEX IF NOT EXISTS idx_activity_source ON activity(source)` and `CREATE INDEX IF NOT EXISTS idx_activity_folder_success ON activity(folder, success)`.

    Define the `runMigrations(db: Database.Database): void` function:
    1. `db.exec('CREATE TABLE IF NOT EXISTS schema_version (version TEXT PRIMARY KEY, applied_at TEXT NOT NULL DEFAULT (datetime(\'now\')))')`.
    2. Query all applied versions into a Set: `db.prepare('SELECT version FROM schema_version').all()`.
    3. Sort migrations by version string (lexicographic — timestamp format ensures correct order).
    4. For each migration not in the applied set, wrap in `db.transaction()`:
       - Call `migration.up(db)`
       - Insert into schema_version: `db.prepare('INSERT INTO schema_version (version) VALUES (?)').run(migration.version)`
    5. Export both `migrations` array and `runMigrations` function.

    **Update `src/log/index.ts`:**

    1. Add import: `import { runMigrations } from './migrations.js';`
    2. In the constructor, replace the `this.migrate()` call with `runMigrations(this.db)`.
    3. Delete the entire `private migrate(): void` method.
    4. The `SCHEMA` constant, constructor, and all other methods remain unchanged.
